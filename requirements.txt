    <script type="text/babel">
        const { useState, useEffect, useRef } = React;

        function PharmaLogixApp() {
            const [view, setView] = useState('login');
            const [token, setToken] = useState(null);
            const [company, setCompany] = useState(null);
            const [error, setError] = useState('');
            const [success, setSuccess] = useState('');

            // Login/Register form states
            const [email, setEmail] = useState('');
            const [password, setPassword] = useState('');
            const [companyName, setCompanyName] = useState('');
            const [isRegistering, setIsRegistering] = useState(false);
            const [loading, setLoading] = useState(false);

            // Chat states
            const [messages, setMessages] = useState([]);
            const [messageInput, setMessageInput] = useState('');
            const [chatLoading, setChatLoading] = useState(false);
            const messagesEndRef = useRef(null);

            // Data upload states
            const [files, setFiles] = useState([]);
            const [uploadLoading, setUploadLoading] = useState(false);
            const fileInputRef = useRef(null);
            const dropZoneRef = useRef(null);

            // Settings states
            const [settingsName, setSettingsName] = useState('');
            const [n8nWebhook, setN8nWebhook] = useState('');
            const [stats, setStats] = useState({ files: 0, messages: 0, data_rows: 0 });
            const [apiKeyRevealed, setApiKeyRevealed] = useState(false);
            const [showApiKey, setShowApiKey] = useState(false);

            const scrollToBottom = () => {
                messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
            };

            useEffect(() => {
                scrollToBottom();
            }, [messages]);

            const showToast = (message, type = 'error') => {
                if (type === 'error') setError(message);
                else setSuccess(message);
                setTimeout(() => {
                    setError('');
                    setSuccess('');
                }, 5000);
            };

            const apiCall = async (endpoint, method = 'GET', body = null) => {
                const options = {
                    method,
                    headers: {
                        'Content-Type': 'application/json'
                    }
                };

                if (token) {
                    options.headers['Authorization'] = `Bearer ${token}`;
                }

                if (body) {
                    options.body = JSON.stringify(body);
                }

                try {
                    const response = await fetch(endpoint, options);
                    if (!response.ok) {
                        const data = await response.json();
                        throw new Error(data.detail || `HTTP ${response.status}`);
                    }
                    return await response.json();
                } catch (err) {
                    throw err;
                }
            };

            const handleLogin = async (e) => {
                e.preventDefault();
                setLoading(true);
                try {
                    const result = await apiCall('/api/auth/login', 'POST', { email, password });
                    setToken(result.token);
                    setCompany(result);
                    setView('chat');
                    setEmail('');
                    setPassword('');
                    showToast('Logged in successfully!', 'success');
                } catch (err) {
                    showToast(err.message);
                } finally {
                    setLoading(false);
                }
            };

            const handleRegister = async (e) => {
                e.preventDefault();
                setLoading(true);
                try {
                    const result = await apiCall('/api/auth/register', 'POST', {
                        company_name: companyName,
                        email,
                        password
                    });
                    setToken(result.token);
                    setCompany(result);
                    setView('chat');
                    setCompanyName('');
                    setEmail('');
                    setPassword('');
                    showToast('Account created successfully!', 'success');
                } catch (err) {
                    showToast(err.message);
                } finally {
                    setLoading(false);
                }
            };

            const handleLogout = () => {
                setToken(null);
                setCompany(null);
                setView('login');
                setMessages([]);
                setFiles([]);
                setError('');
                setSuccess('');
            };

            const handleSendMessage = async (e) => {
                e.preventDefault();
                if (!messageInput.trim()) return;

                const userMessage = messageInput.trim();
                setMessages([...messages, { role: 'user', content: userMessage }]);
                setMessageInput('');
                setChatLoading(true);

                try {
                    const result = await apiCall('/api/chat', 'POST', { message: userMessage });
                    setMessages(prev => [...prev, { role: 'assistant', content: result.response }]);
                    showToast(`Response received (${result.usage} tokens)`, 'success');
                } catch (err) {
                    showToast(err.message);
                } finally {
                    setChatLoading(false);
                }
            };

            const handleFileUpload = async (file) => {
                if (!file) return;

                const allowed = ['.csv', '.xlsx', '.xls'];
                const ext = '.' + file.name.split('.').pop().toLowerCase();

                if (!allowed.includes(ext)) {
                    showToast(`File type ${ext} not supported. Use CSV or Excel.`);
                    return;
                }

                setUploadLoading(true);
                const formData = new FormData();
                formData.append('file', file);

                try {
                    const options = {
                        method: 'POST',
                        headers: {
                            'Authorization': `Bearer ${token}`
                        },
                        body: formData
                    };

                    const response = await fetch('/api/files/upload', options);
                    if (!response.ok) throw new Error('Upload failed');

                    const result = await response.json();
                    setFiles([result, ...files]);
                    showToast('File uploaded successfully!', 'success');
                    if (fileInputRef.current) fileInputRef.current.value = '';
                } catch (err) {
                    showToast(err.message);
                } finally {
                    setUploadLoading(false);
                }
            };

            const handleDeleteFile = async (fileId) => {
                try {
                    await apiCall(`/api/files/${fileId}`, 'DELETE');
                    setFiles(files.filter(f => f.id !== fileId));
                    showToast('File deleted', 'success');
                } catch (err) {
                    showToast(err.message);
                }
            };

            const loadFiles = async () => {
                try {
                    const result = await apiCall('/api/files');
                    setFiles(result);
                } catch (err) {
                    showToast(err.message);
                }
            };

            const clearChatHistory = async () => {
                if (!confirm('Are you sure you want to clear chat history?')) return;
                try {
                    await apiCall('/api/chat/history', 'DELETE');
                    setMessages([]);
                    showToast('Chat history cleared', 'success');
                } catch (err) {
                    showToast(err.message);
                }
            };

            const loadSettings = async () => {
                try {
                    const me = await apiCall('/api/auth/me');
                    setCompany(me);
                    setSettingsName(me.name);
                    setN8nWebhook(me.n8n_webhook);

                    const statsResult = await apiCall('/api/stats');
                    setStats(statsResult);
                } catch (err) {
                    showToast(err.message);
                }
            };

            const handleSaveSettings = async (e) => {
                e.preventDefault();
                try {
                    await apiCall('/api/settings', 'PUT', {
                        company_name: settingsName,
                        n8n_webhook: n8nWebhook
                    });
                    const me = await apiCall('/api/auth/me');
                    setCompany(me);
                    showToast('Settings saved!', 'success');
                } catch (err) {
                    showToast(err.message);
                }
            };

            const copyToClipboard = (text) => {
                navigator.clipboard.writeText(text);
                showToast('Copied to clipboard!', 'success');
            };

            const handleDragOver = (e) => {
                e.preventDefault();
                dropZoneRef.current?.classList.add('dragover');
            };

            const handleDragLeave = () => {
                dropZoneRef.current?.classList.remove('dragover');
            };

            const handleDrop = (e) => {
                e.preventDefault();
                dropZoneRef.current?.classList.remove('dragover');
                if (e.dataTransfer.files.length > 0) {
                    handleFileUpload(e.dataTransfer.files[0]);
                }
            };

            useEffect(() => {
                if (view === 'data' && token) {
                    loadFiles();
                }
            }, [view, token]);

            useEffect(() => {
                if (view === 'settings' && token) {
                    loadSettings();
                }
            }, [view, token]);

            // Login View
            if (!token) {
                return (
                    <div className="auth-container">
                        <div className="auth-card">
                            <div className="auth-header">
                                <div className="auth-logo">💊🚚</div>
                                <h1 className="auth-title">PharmaLogix</h1>
                                <p className="auth-subtitle">Pharmaceutical Logistics AI</p>
                            </div>

                            {error && <div className="error-message">{error}</div>}

                            <form onSubmit={isRegistering ? handleRegister : handleLogin}>
                                {isRegistering && (
                                    <div className="form-group">
                                        <label>Company Name</label>
                                        <input
                                            type="text"
                                            value={companyName}
                                            onChange={(e) => setCompanyName(e.target.value)}
                                            required
                                            placeholder="Your Company Inc."
                                        />
                                    </div>
                                )}

                                <div className="form-group">
                                    <label>Email</label>
                                    <input
                                        type="email"
                                        value={email}
                                        onChange={(e) => setEmail(e.target.value)}
                                        required
                                        placeholder="hello@company.com"
                                    />
                                </div>

                                <div className="form-group">
                                    <label>Password</label>
                                    <input
                                        type="password"
                                        value={password}
                                        onChange={(e) => setPassword(e.target.value)}
                                        required
                                        placeholder="••••••••"
                                    />
                                </div>

                                <button type="submit" className="submit-btn" disabled={loading}>
                                    {loading ? <span className="spinner"></span> : (isRegistering ? 'Create Account' : 'Sign In')}
                                </button>
                            </form>

                            <div className="auth-link">
                                {isRegistering ? (
                                    <>
                                        Already have an account?{' '}
                                        <a onClick={() => setIsRegistering(false)}>Sign In</a>
                                    </>
                                ) : (
                                    <>
                                        Don't have an account?{' '}
                                        <a onClick={() => setIsRegistering(true)}>Create one</a>
                                    </>
                                )}
                            </div>
                        </div>

                        {error && <div className="toast error">{error}</div>}
                        {success && <div className="toast success">{success}</div>}
                    </div>
                );
            }

            // Main App View
            return (
                <div className="container">
                    <div className="sidebar">
                        <div className="logo">
                            <span className="logo-icon">💊</span>
                            <span>PharmaLogix</span>
                        </div>

                        <div className="nav-items">
                            <button className={`nav-item ${view === 'chat' ? 'active' : ''}`} onClick={() => setView('chat')}>
                                💬 Chat
                            </button>
                            <button className={`nav-item ${view === 'data' ? 'active' : ''}`} onClick={() => setView('data')}>
                                📊 Data
                            </button>
                            <button className={`nav-item ${view === 'settings' ? 'active' : ''}`} onClick={() => setView('settings')}>
                                ⚙️ Settings
                            </button>
                            <button className={`nav-item ${view === 'docs' ? 'active' : ''}`} onClick={() => setView('docs')}>
                                🔌 API Docs
                            </button>
                        </div>

                        <div className="company-info">
                            <div className="company-name">{company?.name}</div>
                        </div>

                        <button className="logout-btn" onClick={handleLogout}>
                            Logout
                        </button>
                    </div>

                    <div className="main-content">
                        <div className="content-area">
                            {view === 'chat' && (
                                <div className="chat-container">
                                    <div className="chat-header">
                                        <h2>Chat with PharmaLogix Copilot</h2>
                                        <button className="clear-history-btn" onClick={clearChatHistory}>
                                            Clear History
                                        </button>
                                    </div>

                                    <div className="messages">
                                        {messages.length === 0 ? (
                                            <div className="empty-state">
                                                <div className="empty-state-icon">💬</div>
                                                <p>Welcome to PharmaLogix!</p>
                                                <p>Start by uploading your logistics data, then ask questions about your supply chain.</p>
                                            </div>
                                        ) : (
                                            messages.map((msg, idx) => (
                                                <div key={idx} className={`message ${msg.role}`}>
                                                    <div className="message-bubble">
                                                        {msg.content}
                                                    </div>
                                                </div>
                                            ))
                                        )}
                                        {chatLoading && (
                                            <div className="message assistant">
                                                <div className="message-bubble">
                                                    <span className="spinner"></span>
                                                </div>
                                            </div>
                                        )}
                                        <div ref={messagesEndRef} />
                                    </div>

                                    <form onSubmit={handleSendMessage} className="message-input-area">
                                        <textarea
                                            className="message-input"
                                            value={messageInput}
                                            onChange={(e) => setMessageInput(e.target.value)}
                                            placeholder="Ask about your logistics data, KPIs, compliance, or supply chain..."
                                            rows="2"
                                        />
                                        <button type="submit" className="send-btn" disabled={chatLoading || !messageInput.trim()}>
                                            Send
                                        </button>
                                    </form>
                                </div>
                            )}

                            {view === 'data' && (
                                <div className="data-container">
                                    <div>
                                        <h2 style={{ marginBottom: '16px' }}>Upload Logistics Data</h2>
                                        <div
                                            ref={dropZoneRef}
                                            className="upload-zone"
                                            onDragOver={handleDragOver}
                                            onDragLeave={handleDragLeave}
                                            onDrop={handleDrop}
                                            onClick={() => fileInputRef.current?.click()}
                                        >
                                            <div className="upload-icon">📂</div>
                                            <p className="upload-text">Drag and drop your file here or click to browse</p>
                                            <p className="upload-subtext">Supports CSV and Excel files (.xlsx, .xls)</p>
                                            <input
                                                ref={fileInputRef}
                                                type="file"
                                                className="file-input"
                                                accept=".csv,.xlsx,.xls"
                                                onChange={(e) => {
                                                    if (e.target.files?.[0]) {
                                                        handleFileUpload(e.target.files[0]);
                                                    }
                                                }}
                                            />
                                        </div>
                                    </div>

                                    {uploadLoading && (
                                        <div style={{ textAlign: 'center', padding: '20px' }}>
                                            <span className="spinner"></span> Uploading...
                                        </div>
                                    )}

                                    {files.length > 0 && (
                                        <div>
                                            <h3 style={{ marginBottom: '12px' }}>Uploaded Files ({files.length})</h3>
                                            <div className="files-list">
                                                {files.map(file => (
                                                    <div key={file.id} className="file-item">
                                                        <div className="file-info">
                                                            <div className="file-icon">📄</div>
                                                            <div className="file-details">
                                                                <div className="file-name">{file.name}</div>
                                                                <div className="file-meta">
                                                                    {file.rows} rows • {new Date(file.uploaded_at).toLocaleDateString()}
                                                                </div>
                                                            </div>
                                                        </div>
                                                        <button
                                                            className="delete-btn"
                                                            onClick={() => handleDeleteFile(file.id)}
                                                        >
                                                            Delete
                                                        </button>
                                                    </div>
                                                ))}
                                            </div>
                                        </div>
                                    )}
                                </div>
                            )}

                            {view === 'settings' && (
                                <div className="settings-container">
                                    <h2 style={{ marginBottom: '24px' }}>Settings</h2>

                                    <form onSubmit={handleSaveSettings}>
                                        <div className="settings-section">
                                            <h3>Company Information</h3>

                                            <div className="form-group">
                                                <label>Company Name</label>
                                                <input
                                                    type="text"
                                                    value={settingsName}
                                                    onChange={(e) => setSettingsName(e.target.value)}
                                                    placeholder="Your Company"
                                                />
                                            </div>

                                            <div className="form-group">
                                                <label>Email</label>
                                                <input
                                                    type="email"
                                                    value={company?.email}
                                                    disabled
                                                    style={{ opacity: 0.5 }}
                                                />
                                            </div>

                                            <div className="form-group">
                                                <label>API Key</label>
                                                <div className="api-key-display">
                                                    <div className="api-key-value">
                                                        {apiKeyRevealed ? company?.api_key : '••••••••••••••••••••••'}
                                                    </div>
                                                    <button
                                                        type="button"
                                                        className="copy-btn"
                                                        onClick={() => {
                                                            if (!apiKeyRevealed) setApiKeyRevealed(true);
                                                            copyToClipboard(company?.api_key);
                                                        }}
                                                    >
                                                        {apiKeyRevealed ? 'Copy' : 'Reveal'}
                                                    </button>
                                                </div>
                                            </div>
                                        </div>

                                        <div className="settings-section">
                                            <h3>Email Integration</h3>
                                            <div className="form-group">
                                                <label>n8n Webhook URL</label>
                                                <input
                                                    type="text"
                                                    value={n8nWebhook}
                                                    onChange={(e) => setN8nWebhook(e.target.value)}
                                                    placeholder="https://your-n8n-instance.app.n8n.cloud/webhook/..."
                                                />
                                            </div>
                                            <p style={{ fontSize: '12px', color: '#9ca3af', marginTop: '8px' }}>
                                                Optional: Add your n8n webhook URL to enable email reports from the copilot
                                            </p>
                                        </div>

                                        <button type="submit" className="submit-btn" style={{ width: '100%' }}>
                                            Save Settings
                                        </button>
                                    </form>

                                    <div className="settings-section">
                                        <h3>Usage Statistics</h3>
                                        <div className="stats-grid">
                                            <div className="stat-card">
                                                <div className="stat-value">{stats.files}</div>
                                                <div className="stat-label">Files Uploaded</div>
                                            </div>
                                            <div className="stat-card">
                                                <div className="stat-value">{stats.messages}</div>
                                                <div className="stat-label">Messages Sent</div>
                                            </div>
                                            <div className="stat-card">
                                                <div className="stat-value">{stats.data_rows.toLocaleString()}</div>
                                                <div className="stat-label">Data Rows</div>
                                            </div>
                                        </div>
                                    </div>
                                </div>
                            )}

                            {view === 'docs' && (
                                <div className="docs-container">
                                    <h2 style={{ marginBottom: '24px' }}>API Documentation</h2>

                                    <div className="docs-section">
                                        <h3>Authentication</h3>
                                        <p className="docs-text">
                                            All API requests require a Bearer token in the Authorization header:
                                        </p>
                                        <div className="code-block">
Authorization: Bearer YOUR_JWT_TOKEN
                                        </div>
                                        <p className="docs-text" style={{ marginTop: '12px' }}>
                                            Or use your API key:
                                        </p>
                                        <div className="code-block">
X-API-Key: plx_your_api_key_here
                                        </div>
                                    </div>

                                    <div className="docs-section">
                                        <h3>Endpoints</h3>
                                        <h4>POST /api/chat</h4>
                                        <p className="docs-text">Send a message to the PharmaLogix copilot</p>
                                        <div className="code-block">
{`{
  "message": "What are our KPIs this quarter?"
}`}
                                        </div>

                                        <h4>POST /api/files/upload</h4>
                                        <p className="docs-text">Upload a logistics data file (multipart/form-data)</p>

                                        <h4>GET /api/files</h4>
                                        <p className="docs-text">List all uploaded files</p>

                                        <h4>POST /api/email/send</h4>
                                        <p className="docs-text">Send an email via your n8n webhook</p>
                                    </div>

                                    <div className="docs-section">
                                        <h3>Example: Python</h3>
                                        <div className="code-block">
{`import requests

headers = {
    "Authorization": "Bearer YOUR_TOKEN",
    "Content-Type": "application/json"
}

response = requests.post(
    "https://your-app.com/api/chat",
    json={"message": "Analyze Q1 data"},
    headers=headers
)
print(response.json())`}
                                        </div>
                                    </div>

                                    <div className="docs-section">
                                        <h3>Example: cURL</h3>
                                        <div className="code-block">
{`curl -X POST https://your-app.com/api/chat \\
  -H "Authorization: Bearer YOUR_TOKEN" \\
  -H "Content-Type: application/json" \\
  -d '{"message":"What is OTIF performance?"}'`}
                                        </div>
                                    </div>
                                </div>
                            )}
                        </div>
                    </div>

                    {error && <div className="toast error">{error}</div>}
                    {success && <div className="toast success">{success}</div>}
                </div>
            );
        }

        ReactDOM.createRoot(document.getElementById('root')).render(<PharmaLogixApp />);
    </script>
